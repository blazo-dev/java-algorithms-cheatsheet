HashMap
O(1) avg 
HashMap<Integer, Integer> map = new HashMap<>();

// CRUD básico
map.put(key, value);
map.get(key);                     // null si no existe
map.getOrDefault(key, 0);        // default si no existe
map.containsKey(key);
map.containsValue(val);
map.remove(key);
map.size();
map.isEmpty();
map.clear();

// Actualizar valores
map.put(key, map.getOrDefault(key, 0) + 1);  // contador
map.putIfAbsent(key, value);      // solo si no existe
map.replace(key, newValue);       // solo si existe
map.merge(key, 1, Integer::sum);  // putIfAbsent + combinar
map.compute(key, (k,v) -> v == null ? 1 : v+1);

// Iterar
for (Map.Entry<Integer,Integer> e : map.entrySet()) {
    e.getKey();   e.getValue();
}
for (int k : map.keySet()) { }
for (int v : map.values()) { }

// Otras utilidades
map.getOrDefault(key, defaultVal);
map.forEach((k, v) -> process(k, v));
map.entrySet().stream()
   .max(Map.Entry.comparingByValue()); // entrada con valor máximo

Stack / Queue (Deque)
O(1) 
// STACK (LIFO) — usa Deque, no Stack
Deque<Integer> stack = new ArrayDeque<>();
stack.push(val);          // agregar arriba  (addFirst)
stack.pop();              // sacar arriba    (removeFirst)
stack.peek();             // ver sin sacar   (peekFirst)
stack.isEmpty();
stack.size();

// QUEUE (FIFO)
Queue<Integer> q = new LinkedList<>();
q.offer(val);             // agregar atrás  (no lanza excepción)
q.poll();                 // sacar frente   (null si vacío)
q.peek();                 // ver sin sacar  (null si vacío)
q.isEmpty();
q.size();

// DEQUE como doble cola (muy útil en sliding window)
Deque<Integer> dq = new ArrayDeque<>();
dq.addFirst(val);   dq.addLast(val);
dq.removeFirst();   dq.removeLast();
dq.peekFirst();     dq.peekLast();
Regla: Stack → push/pop/peek · Queue → offer/poll/peek

PriorityQueue (Heap)
O(log n) 
// Min-heap (default) — menor sale primero
PriorityQueue<Integer> minH = new PriorityQueue<>();

// Max-heap — mayor sale primero
PriorityQueue<Integer> maxH =
    new PriorityQueue<>(Collections.reverseOrder());

// Heap de arrays (por ejemplo, por primer elemento)
PriorityQueue<int[]> pq =
    new PriorityQueue<>((a, b) -> a[0] - b[0]);

minH.offer(val);          // agregar — O(log n)
minH.poll();              // sacar el mínimo — O(log n)
minH.peek();              // ver sin sacar — O(1)
minH.contains(val);       // O(n) — lento
minH.size();
minH.isEmpty();

// Patrón K elementos más pequeños
for (int n : nums) {
    maxH.offer(n);
    if (maxH.size() > k) maxH.poll(); // elimina el mayor
}
Cuándo usarlo: "encuentra los K elementos más grandes/pequeños"

TreeMap
O(log n) sorted 
// Como HashMap pero con claves ORDENADAS
TreeMap<Integer, String> tm = new TreeMap<>();

tm.put(key, value);
tm.get(key);
tm.containsKey(key);
tm.remove(key);

// Lo que hace especial a TreeMap
tm.firstKey();            // clave más pequeña
tm.lastKey();             // clave más grande
tm.floorKey(k);           // mayor clave <= k
tm.ceilingKey(k);         // menor clave >= k
tm.lowerKey(k);           // mayor clave < k (estricto)
tm.higherKey(k);          // menor clave > k (estricto)
tm.headMap(k);            // submap con claves < k
tm.tailMap(k);            // submap con claves >= k
tm.subMap(lo, hi);        // submap [lo, hi)

// Iterar en orden
for (Map.Entry<Integer,String> e : tm.entrySet()) { }
// Iterar en orden inverso
tm.descendingKeySet();
Cuándo usarlo: necesitas el vecino más cercano a un valor, o iterar en orden.

TreeSet
O(log n) sorted 
// Como HashSet pero con elementos ORDENADOS
TreeSet<Integer> ts = new TreeSet<>();

ts.add(val);
ts.contains(val);
ts.remove(val);
ts.size();

// Navegación (la ventaja sobre HashSet)
ts.first();               // elemento más pequeño
ts.last();                // elemento más grande
ts.floor(val);            // mayor elemento <= val
ts.ceiling(val);          // menor elemento >= val
ts.lower(val);            // mayor elemento < val (estricto)
ts.higher(val);           // menor elemento > val (estricto)
ts.pollFirst();           // saca y devuelve el menor
ts.pollLast();            // saca y devuelve el mayor
ts.headSet(val);          // elementos < val
ts.tailSet(val);          // elementos >= val
ts.subSet(lo, hi);        // elementos en [lo, hi)

// Iterar en orden
for (int x : ts) { }      // ascendente
ts.descendingSet();       // vista en orden inverso
Cuándo usarlo: necesitas el elemento más cercano a un valor en un conjunto.

LinkedList<Integer> ll = new LinkedList<>();

// Como List

Binary Search
O(log n) 
// Built-in (array debe estar ordenado)
int idx = Arrays.binarySearch(nums, target);
// Retorna índice si existe, negativo si no

// Implementación manual (más útil en entrevistas)
int left = 0, right = nums.length - 1;
while (left <= right) {
    int mid = left + (right - left) / 2;  // evita overflow
    if (nums[mid] == target)      return mid;
    else if (nums[mid] < target)  left  = mid + 1;
    else                          right = mid - 1;
}
return -1;
